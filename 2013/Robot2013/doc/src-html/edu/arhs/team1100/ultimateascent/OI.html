014/**
015 * This class is the glue that binds the controls on the physical operator
016 * interface to the commands and command groups that allow control of the robot.
017 */
018public class OI {
019
020    private static OI instance;
021
022    private AttackThree leftStick;
023    private AttackThree rightStick;
024    private XboxController xbox;
025
026    RecordCommand recorder;
027
028    /**
029     * Gets the current instance.
030     * @return instance of OI
031     */
032    public static OI getInstance(){
033        if(instance == null){
034            instance = new OI();
035        }
036        return instance;
037    }
038
039    /**
040     * Constructor
041     */
042    public OI(){
043        leftStick = new AttackThree(RobotMap.C_LEFT_JOYSTICK_CHANNEL, 0.1);
044        rightStick = new AttackThree(RobotMap.C_RIGHT_JOYSTICK_CHANNEL, 0.1);
045        xbox = new XboxController(RobotMap.C_XBOX_CONTROLLER_CHANNEL, 0.1);
046        //drive controls
047        leftStick.getButton(RobotMap.C_TOGGLE_DRIVE).whenPressed(new ToggleDriveModeCommand());
048        leftStick.getButton(RobotMap.C_CAMERATEST).whileHeld(new CameraTestCommand());
049
050        rightStick.getButton(RobotMap.C_CALIBRATE_GYRO).whenPressed(new CalibrateGyroCommand());
051        rightStick.getButton(RobotMap.C_DRIVE_PID).whileHeld(new JoystickPIDMecanumCommand());
052        leftStick.getButton(RobotMap.C_DRIVE_CAMERA).whileHeld(new CameraPIDMecanumCommand());
053        rightStick.getButton(RobotMap.C_STOP_DRIVE).whenPressed(new StopDriveCommand(0.1));
054
055        //Recording Command Stuff
056        recorder = new RecordCommand(20); //interval is 20 ms
057        rightStick.getButton(RobotMap.C_RECORD).whileHeld(recorder);
058        rightStick.getButton(RobotMap.C_PLAY_RECORDING).whenPressed(new PlayRecordingCommand(recorder));
059        rightStick.getButton(RobotMap.C_PRINT_RECORDING).whenPressed(new PrintRecordingCodeCommand(recorder));
060
061        rightStick.getButton(8).whenPressed(new SquareWeaveRoutine());
062        //xbox Stuff
063        xbox.getButtonRightBumper().whileHeld(new ShootFrisbeeCommand());
064        xbox.getButtonLeftBumper().whileHeld(new ShootAllFrisbeesCommand());
065
066
067
068    }
069
070    /**
071     * Get the left joystick
072     * @return left AttackThree object
073     */
074    public AttackThree getLeftJoystick(){
075        return leftStick;
076    }
077
078    /**
079     * Get right joystick
080     * @return right AttackThree
081     */
082    public AttackThree getRightJoystick(){
083        return rightStick;
084    }
085
086    /**
087     * Get xbox controller
088     * @return XboxController object
089     */
090    public XboxController getXboxController(){
091        return xbox;
092    }
093
094    //// CREATING BUTTONS
095    // One type of button is a joystick button which is any button on a joystick.
096    // You create one by telling it which joystick it's on and which button
097    // number it is.
098    // Joystick stick = new Joystick(port);
099    // Button button = new JoystickButton(stick, buttonNumber);
100
101    // Another type of button you can create is a DigitalIOButton, which is
102    // a button or switch hooked up to the cypress module. These are useful if
103    // you want to build a customized operator interface.
104    // Button button = new DigitalIOButton(1);
105
106    // There are a few additional built in buttons you can use. Additionally,
107    // by subclassing Button you can create custom triggers and bind those to
108    // commands the same as any other Button.
109
110    //// TRIGGERING COMMANDS WITH BUTTONS
111    // Once you have a button, it's trivial to bind it to a button in one of
112    // three ways:
113
114    // Start the command when the button is pressed and let it run the command
115    // until it is finished as determined by it's isFinished method.
116    // button.whenPressed(new ExampleCommand());
117
118    // Run the command while the button is being held down and interrupt it once
119    // the button is released.
120    // button.whileHeld(new ExampleCommand());
121
122    // Start the command the button is released  and let it run the command
123    // until it is finished as determined by it's isFinished method.
124    // button.whenReleased(new ExampleCommand());
125}
