001package edu.arhs.team1100.ultimateascent.recording;
002
003import edu.arhs.team1100.ultimateascent.OI;
004import edu.arhs.team1100.ultimateascent.commands.CommandBase;
005import edu.arhs.team1100.ultimateascent.subsystems.DriveSubsystem;
006import edu.arhs.team1100.ultimateascent.util.Log;
007import edu.wpi.first.wpilibj.Joystick;
008import java.util.Vector;
009
010/**
011 * Records robot movements for later replay
012 * @author Team 1100
013 */
014public class RecordCommand extends CommandBase {
015
016    private Vector recording;
017    private int interval = 0;
018    private long last = 0;
019
020    public RecordCommand(int hz) {
021        recording = new Vector();
022        interval = 1000 / hz;
023    }
024
025    protected void initialize() {
026        Log.log(this, "START RECORD", Log.LEVEL_DEBUG);
027        recording = new Vector();
028        last = System.currentTimeMillis();
029
030    }
031
032    protected void execute() {
033        long t = System.currentTimeMillis();
034        if (t - last >= interval) {
035            double X = OI.getInstance().getLeftJoystick().getAxis(Joystick.AxisType.kX);
036            double Y = OI.getInstance().getLeftJoystick().getAxis(Joystick.AxisType.kY);
037            double R = OI.getInstance().getRightJoystick().getAxis(Joystick.AxisType.kX);
038            int mode = DriveSubsystem.getInstance().getDriveMode();
039            recording.addElement(new ControllerState(X, Y, R, mode));
040            last = t;
041        }
042    }
043
044    /**
045     * @return the last recording by this recorded 
046     */
047    public Vector getRecording() {
048        return recording;
049    }
050
051    /**
052     * 
053     * @return the frequency at which recordings are made
054     */
055    public int getInterval() {
056        return interval;
057    }
058
059    protected boolean isFinished() {
060        return false;
061    }
062
063    protected void end() {
064    }
065
066    protected void interrupted() {
067    }
068}
