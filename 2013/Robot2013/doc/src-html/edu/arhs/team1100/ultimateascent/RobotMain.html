017import edu.arhs.team1100.ultimateascent.util.Log;
018import edu.wpi.first.wpilibj.IterativeRobot;
019import edu.wpi.first.wpilibj.command.Command;
020import edu.wpi.first.wpilibj.command.Scheduler;
021import edu.wpi.first.wpilibj.livewindow.LiveWindow;

023/**
024 * The VM is configured to automatically run this class, and to call the
025 * functions corresponding to each mode, as described in the IterativeRobot
026 * documentation. If you change the name of this class or the package after
027 * creating this project, you must also update the manifest file in the resource
028 * directory.
029 */
030public class RobotMain extends IterativeRobot {
031
032    Command autonomousCommand;
033    long lastTime = 0;
034    long totalTime = 0;
035    long cycles = 0;
036
037    /**
038     * This function is run when the robot is first started up and should be
039     * used for any initialization code.
040     */
041    public void robotInit() {
042        //initialize the Log utility
043        Log.init();
044        Log.setMinLevel(Log.LEVEL_DEBUG);
045        //add all logging classes
046        Log.addClass(RobotMain.class, Log.LEVEL_DEBUG);
047        //Log.addClass(DriveSubsystem.class, Log.LEVEL_DEBUG);
048        Log.addClass(CalibrateGyroCommand.class, Log.LEVEL_OFF);
049        Log.addClass(JoystickPIDMecanumCommand.class, Log.LEVEL_OFF);
050        Log.addClass(RecordCommand.class, Log.LEVEL_DEBUG);
051        Log.addClass(PlayRecordingCommand.class, Log.LEVEL_DEBUG);
052        Log.addClass(CameraTestCommand.class, Log.LEVEL_DEBUG);
053        Log.addClass(Camera.class, Log.LEVEL_DEBUG);
054        // instantiate the command used for the autonomous period
055
056        // Initialize all subsystems
057        CommandBase.init();
058    }
059
060    public void autonomousInit() {
061        // schedule the autonomous command (example)
062        autonomousCommand.start();
063    }
064
065    /**
066     * This function is called periodically during autonomous
067     */
068    public void autonomousPeriodic() {
069        Scheduler.getInstance().run();
070    }
071
072    public void teleopInit() {
073        // This makes sure that the autonomous stops running when
074        // teleop starts running. If you want the autonomous to
075        // continue until interrupted by another command, remove
076        // this line or comment it out.
077        lastTime = System.currentTimeMillis();
078        Scheduler.getInstance().enable();
079    }
080
081    /**
082     * This function is called periodically during operator control
083     */
084    public void teleopPeriodic() {
085        Scheduler.getInstance().run();
086        printRate();
087
088
089    }
090
091    private void printRate() {
092
093        long curTime = System.currentTimeMillis();
094        long d = curTime - lastTime;
095        totalTime += d;
096        cycles++;
097
098        if (totalTime >= 1000) {
099            DSLog.log(3, "Rate: " + (cycles));
100            cycles = 0;
101            totalTime = 0;
102        }
103
104        lastTime = curTime;
105
106    }
107
108    /**
109     * This function is called periodically during test mode
110     */
111    public void testPeriodic() {
112        LiveWindow.run();
113    }
114
115    public void disabledInit() {
116        Scheduler.getInstance().removeAll();
117        Scheduler.getInstance().disable();
118    }
119}
