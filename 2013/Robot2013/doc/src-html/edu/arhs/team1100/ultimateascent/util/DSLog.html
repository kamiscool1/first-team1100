010/**
011 * Logs messages to the Driver Station
012 */
013public class DSLog {
014
015    // going to treat the LCD as a stack, in reverse here because
016    // new lines appear on the bottom and scroll upwards (mostly)
017    static DriverStationLCD.Line[] lcdLines = {
018        DriverStationLCD.Line.kUser6, // bottom line
019        DriverStationLCD.Line.kUser5,
020        DriverStationLCD.Line.kUser4,
021        DriverStationLCD.Line.kUser3,
022        DriverStationLCD.Line.kUser2,
023        DriverStationLCD.Line.kMain6 // top line
024    };
025    static String[] lcdHistory = new String[6];  // array to scroll lcd display
026    static String clearLine = "                     "; // 21 spaces to clear a line
027
028    public DSLog() {
029        for (int i = 0; i < lcdHistory.length; i++) {
030            lcdHistory[i] = "";
031        }
032    }
033
034    /**
035     * When run, the class starts a continuous loop to update the dashboard
036     * constantly. This is probably a bad way to do it.
037     */
038    public void step() {
039        updateDashboard();
040    }

042    /**
043     * Messages to the LCD must be 21 characters long or you get artifacts from
044     * previous lines. This should pad the desired message out to the proper
045     * length.
046     *
047     * @param msg the message to be padded out
048     * @return line the message line padded out to 21 characters
049     */
050    static private String padMsg(String msg) {
051        String line = msg;
052        if (line.length() < 21) {
053            for (int i = msg.length() + 1; i <= 21; i++) {
054                line += " ";
055            }
056        }
057        line = line.substring(0, 21);
058        return line;
059    }
060
061    /**
062     * Clears the Driver Station LCD
063     */
064    public static void clearLCD() {
065        for (int i = 0; i < 6; i++) {
066            lcdHistory[i] = clearLine;
067            DriverStationLCD.getInstance().println(lcdLines[i], 1, lcdHistory[i]);
068        }
069        DriverStationLCD.getInstance().updateLCD();
070    }
071
072    /**
073     * This makes it easier to send message to a specific line. Takes the
074     * desired line number (top to bottom, 1-6) and the message to print as
075     * arguments.
076     *
077     * @param line which line of the LCD the message should be displayed on
078     * @param msg the message to be displayed
079     */
080    public static void log(int line, String msg) {
081
082        int index = 6 - line; // array is set up bottom to top, this reverses it
083        DriverStationLCD.getInstance().println(lcdLines[index], 1, padMsg(msg));
084        DriverStationLCD.getInstance().updateLCD(); // keep forgetting this, d'oh!
085    } // end method toLCDLine()

087    /**
088     * Taken from DashboardExampleProject. I'll try to explain how this is
089     * organized.
090     *
091     * sThe LabView dashboard is organized with all the various displays in
092     * "clusters". These clusters are embedded in other clusters, which are
093     * embedded in otherwise, and so on. For example, PWM display 1 is embedded
094     * in the cluster for Digital Module 0, which is embedded in Digital Cluster
095     * 1, which is in the cluster containing all the digital modules, which is
096     * in the cluster containing everything. Whew. Basically it's a tree
097     * structure.
098     *
099     * To send information to the dashboard, you have to "pack" this tree,
100     * filling in all the various clusters with the correct data, and then send
101     * the whole tree at one time to the dashboard. You create the tree by using
102     * the addCluster() method, nesting each lower level of the tree in the
103     * higher ones. When a cluster is loaded you finalize it, which basically
104     * packs up that particular module for shipment to the dashboard. After all
105     * the clusters are finalized you use the commit() method to send it to the
106     * dashboard.
107     *
108     * I've extracted the tree structure from the LabView dashboard code to
109     * interpret what's below. I'll try to provide enough comments for you to
110     * understand which module is where, and which instrument display it
111     * connects to.
112     *
113     * To keep the dashboard up to date you have to call this method repeatedly,
114     * either in an infinite loop or in teleopContinuous() if you use
115     * IterativeRobot. I use an infinite loop running in a separate thread.
116     */
117    private void updateDashboard() {
118
119        // first, get an instance of the dashboard "packer" file.  This is
120        // essentially a map of how the dashboard has the clusters configured.
121        Dashboard lowDashData = DriverStation.getInstance().getDashboardPackerLow();
122
123        // We now recreate the tree using the addCluster method, nesting
124        // the individual display panels.
125
126        // First, the overall container that holds everything else
127        lowDashData.addCluster();  // overall container
128        {
129            // There are 3 clusters in the next level, the analog modules,
130            // the digital modules, and the solenoids.
131
132            // Add the cluster containing the analog modules.
133            lowDashData.addCluster();
134            {
135                // The analog module cluster contains two analog modules, one
136                // for each of the first two slots on the cRio.
137
138                // Add the cluster for Analog Module 0 (Slot One on the cRio)
139                lowDashData.addCluster();
140
141
142                {
143                    // Here we just iterate through the analog ports, adding
144                    // float values containing the average voltage for the given
145                    // port.
146                    for (int i = 1; i <= 8; i++) {
147                        lowDashData.addFloat((float) AnalogModule.getInstance(1).getAverageVoltage(i));
148                    }
149                }
150                // Finished with Analog Module 0, finalize the cluster
151                lowDashData.finalizeCluster(); // finalize analog module 0
152
153                /// Add the cluster for Analog Module 1 (Slot Two on the cRio)
154                lowDashData.addCluster();
155                {
156                    // And again, iterate through the voltages for each port
157                    // on the module
158                    for (int i = 1; i <= 8; i++) {
159                        lowDashData.addFloat((float) AnalogModule.getInstance(2).getAverageVoltage(i));
160                    }
161                }
162                // Finished with Analog Module 1, finalize the cluster.
163                lowDashData.finalizeCluster(); // finalize analog module 1
164            }
165            // Done loading the data for the analog modules, so we finalize
166            // the cluster that holds both of them.
167            lowDashData.finalizeCluster(); // finalize analog cluster
168
169            // Add the cluster containing the digital sub clusters.
170            lowDashData.addCluster();
171            {
172                // The digital cluster contains two sub clusters, one for
173                // slot 4 and one for slot 6 on the cRio.  Each of those
174                // clusters contain modules for relays and for the PWM
175                // slots.
176
177                // Add the cluster for Digital SubCluster 1 (slot 4 on the cRio)
178                lowDashData.addCluster();
179                {
180                    // The digital module itself contains two sub modules.
181                    // The numbering system is screwey, but it's taken straight
182                    // from the LabView code...
183
184                    // Add the cluster Digital Module 0 in SubCluster 1
185                    lowDashData.addCluster();
186                    {
187                        int module = 4; // this specifies the cRio slot
188
189                        // And the next 4 lines handle relays
190                        lowDashData.addByte(DigitalModule.getInstance(module).getRelayForward());
191                        lowDashData.addByte(DigitalModule.getInstance(module).getRelayForward());
192                        lowDashData.addShort(DigitalModule.getInstance(module).getAllDIO());
193                        lowDashData.addShort(DigitalModule.getInstance(module).getDIODirection());
194
195                        // Add the cluster for PWM module 1 - I know for a fact
196                        // that these are the 10 PWM connections on the digital
197                        // sidecar hooked to the module in cRio Slot 4 :)
198                        lowDashData.addCluster();
199                        {
200                            // Iterate through the PWM values for each port
201                            for (int i = 1; i <= 10; i++) {
202                                lowDashData.addByte((byte) DigitalModule.getInstance(module).getPWM(i));
203                            }
204                        }
205                        // Finished with PWM Module 1, finalize the cluster.
206                        lowDashData.finalizeCluster(); // finalize PWM 1
207                    }
208                    // And finished with Digital Module 0
209                    lowDashData.finalizeCluster(); // finalize digital module 0
210                }
211                // And with Digital SubCluster 1
212                lowDashData.finalizeCluster(); // finalize subcluster 1
213
214                // Add the cluster for Digital SubCluster 2 (slot 6 on the cRio)
215                lowDashData.addCluster();
216                {
217                    // Add the cluster for Digital Module 1 in SubCluster 2
218                    lowDashData.addCluster();
219                    {
220                        int module = 6; // cRio slot 6
221
222                        // And again, load the relay data
223                        lowDashData.addByte(DigitalModule.getInstance(module).getRelayForward());
224                        lowDashData.addByte(DigitalModule.getInstance(module).getRelayReverse());
225                        lowDashData.addShort(DigitalModule.getInstance(module).getAllDIO());
226                        lowDashData.addShort(DigitalModule.getInstance(module).getDIODirection());
227
228                        // Add the cluster for PWM Module 2
229                        lowDashData.addCluster();
230                        {
231                            // iterate through all 10 PWM ports
232                            for (int i = 1; i <= 10; i++) {
233                                lowDashData.addByte((byte) DigitalModule.getInstance(module).getPWM(i));
234                            }
235                        }
236                        // Finished with PWM Module 2, finalize it.
237                        lowDashData.finalizeCluster(); // finalize PWM 2
238                    }
239                    // Finished with Digital Module 2
240                    lowDashData.finalizeCluster(); // finalize digital 2
241                }
242                // And done with Digital SubCluster 2
243                lowDashData.finalizeCluster(); // finalize subcluster 2
244            }
245            // And that completes the entire Digital Cluster
246            lowDashData.finalizeCluster(); // finalize digital cluster
247
248            // The solenoids, for some reason, are not in a cluster of their
249            // own, but are instead sitting in the overall container.  As such,
250            // we don't need to addCluster() for them, we just need to add
251            // the data.
252            lowDashData.addByte(Solenoid.getAllFromDefaultModule());
253
254            // If you are going to add instruments to the dashboard the
255            // easiest place to put them is in the overall cluster, and then
256            // load the values for them here the same way the solenoid was
257            // done.  Makes me think the solenoid was an afterthought.
258            // Actually I think the whole dashboard was designed haphazardly
259            // instead of in a planned manner, which is why this whole structure
260            // sucks so bad.
261
262        }
263
264        // Once all the inputs are read and loaded, we close the overall container
265        lowDashData.finalizeCluster(); // finalize entire tree structure.
266
267        // We are now ready to send our packed up data tree to the dashboard
268        // for display.
269        lowDashData.commit(); // commit changes to update dashboard
270
271    } // end method updateDashboard()
