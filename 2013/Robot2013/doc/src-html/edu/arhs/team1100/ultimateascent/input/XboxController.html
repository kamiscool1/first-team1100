001package edu.arhs.team1100.ultimateascent.input;
002
003import edu.wpi.first.wpilibj.Joystick;
004import edu.wpi.first.wpilibj.buttons.JoystickButton;
005
006/**
007 *
008 * @author Team 1100
009 */
010
011public class XboxController extends Joystick {
012
013    private JoystickButton buttonA;
014    private JoystickButton buttonB;
015    private JoystickButton buttonX;
016    private JoystickButton buttonY;
017    private JoystickButton buttonLeftBumper;
018    private JoystickButton buttonRightBumper;
019    private JoystickButton buttonBack;
020    private JoystickButton buttonStart;
021    private JoystickButton buttonLeftStick;
022    private JoystickButton buttonRightStick;
023    
024    private double joystickDeadband = 0.1;
025    public XboxController(int channel, double joystickDeadband) {        
026        super(channel);
027
028        buttonA = new JoystickButton(this, 1);
029        buttonB = new JoystickButton(this, 2);
030        buttonX = new JoystickButton(this, 3);
031        buttonY = new JoystickButton(this, 4);
032        buttonLeftBumper = new JoystickButton(this, 5);
033        buttonRightBumper = new JoystickButton(this, 6);
034        buttonBack = new JoystickButton(this, 7);
035        buttonStart = new JoystickButton(this, 8);
036        buttonLeftStick = new JoystickButton(this, 9);
037        buttonRightStick = new JoystickButton(this, 10);
038        
039        this.joystickDeadband = joystickDeadband;
040    }
041
042    public JoystickButton getButtonA() {
043        return buttonA;
044    }
045
046    public JoystickButton getButtonB() {
047        return buttonB;
048    }
049
050    public JoystickButton getButtonX() {
051        return buttonX;
052    }
053
054    public JoystickButton getButtonY() {
055        return buttonY;
056    }
057
058    public JoystickButton getButtonLeftBumper() {
059        return buttonLeftBumper;
060    }
061
062    public JoystickButton getButtonRightBumper() {
063        return buttonRightBumper;
064    }
065
066    public JoystickButton getButtonBack() {
067        return buttonBack;
068    }
069
070    public JoystickButton getButtonStart() {
071        return buttonStart;
072    }
073    
074    public JoystickButton getButtonLeftStick() {
075        return buttonLeftStick;
076    }
077
078    public JoystickButton getButtonRightStick() {
079        return buttonRightStick;
080    }
081
082    public double getAxis(Joystick.AxisType axis) {
083        double val = super.getAxis(axis);
084        if(Math.abs(val) <= joystickDeadband) {
085            val = 0.0;
086        }        
087        return val;
088    }
089}
