001package edu.arhs.team1100.ultimateascent.autonomous;
002
003import edu.arhs.team1100.ultimateascent.subsystems.DriveSubsystem;
004import edu.wpi.first.wpilibj.command.CommandGroup;
005
006/**
007 *
008 * @author Team1100
009 */
010public class DriveInACircleCommandGroup extends CommandGroup {
011
012    /**
013     * Makes the robot move in a circle
014     * @param speed speed of wheels
015     * @param duration total duration of entire movement in second
016     * @param rotations number of how many times the angle moves e.g. input 8 would make a octagon
017     */
018    public DriveInACircleCommandGroup(double speed, double duration, double rotations) {
019
020        for (int angle = 0; angle <= 360; angle += 360 / rotations) {
021            addSequential(new DriveInALineCommand(speed, angle, duration / rotations));
022        }
023        addSequential(new StopDriveCommand(0.5));
024
025    }
026}
