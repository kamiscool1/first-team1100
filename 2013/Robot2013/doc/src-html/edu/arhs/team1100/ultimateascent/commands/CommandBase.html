001package edu.arhs.team1100.ultimateascent.commands;
002
003import edu.arhs.team1100.ultimateascent.OI;
004import edu.arhs.team1100.ultimateascent.subsystems.DriveSubsystem;
005import edu.arhs.team1100.ultimateascent.sensors.Camera;
006import edu.wpi.first.wpilibj.command.Command;
007
008/**
009 * The base for all commands. All atomic commands should subclass CommandBase.
010 * CommandBase stores creates and stores each control system. To access a
011 * subsystem elsewhere in your code in your code use CommandBase.exampleSubsystem
012 * @author Team 1100
013 */
014public abstract class CommandBase extends Command {
015
016    public static void init() {
017        // This MUST be here. If the OI creates Commands (which it very likely
018        // will), constructing it during the construction of CommandBase (from
019        // which commands extend), subsystems are not guaranteed to be
020        // yet. Thus, their requires() statements may grab null pointers. Bad
021        // news. Don't move it.
022        DriveSubsystem.getInstance();
023        Camera.getInstance();
024        OI.getInstance();
025
026        // Show what command your subsystem is running on the SmartDashboard
027        //SmartDashboard.putData(exampleSubsystem);
028    }
029
030    public CommandBase(String name) {
031        super(name);
032    }
033
034    public CommandBase() {
035        super();
036    }
037}
