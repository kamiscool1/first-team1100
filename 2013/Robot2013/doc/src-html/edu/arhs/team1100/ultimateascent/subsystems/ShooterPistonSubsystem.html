001package edu.arhs.team1100.ultimateascent.subsystems;
002
003import edu.arhs.team1100.ultimateascent.RobotMap;
004import edu.wpi.first.wpilibj.DigitalInput;
005import edu.wpi.first.wpilibj.command.Subsystem;
006import edu.wpi.first.wpilibj.Solenoid;
007
008/**
009 *
010 * @author Team 1100
011 */
012public class ShooterPistonSubsystem extends Subsystem {
013
014    private Solenoid shooterPiston;
015    static ShooterPistonSubsystem instance;
016    
017    
018    private DigitalInput limitSwitch;
019    private boolean lastLimitState;
020    
021    private int frisbeeCount = 0;
022    
023    public ShooterPistonSubsystem() {
024        shooterPiston = new Solenoid(RobotMap.S_SOLENOID_SHOOTER_PISTON);
025        
026        limitSwitch = new DigitalInput(RobotMap.S_FRISBEE_LIMIT_SWITCH);
027
028    }
029    
030    public static ShooterPistonSubsystem getInstance() {
031        if(instance == null) {
032            instance = new ShooterPistonSubsystem();
033            instance.initDefaultCommand();
034        }
035        return instance;
036    }
037
038    /**
039     * Shoots one frisbee
040     */
041    public void shoot() {
042        shooterPiston.set(true);
043        shooterPiston.set(false);
044        frisbeeCount--;
045    }
046    
047    /**
048     * Counts frisbees
049     */
050    public void updateFrisbeeCount(){
051        boolean limitState = limitSwitch.get();
052        if(limitState && !lastLimitState){ //if switch is open, and was closed before
053            frisbeeCount++;
054        }
055        lastLimitState = limitState;
056        
057    }
058
059    /**
060     * 
061     * @return number of frisbees being carried
062     */
063    public int getFrisbeeCount() {
064        return frisbeeCount;
065    }
066    protected void initDefaultCommand() {
067    }
068
069}
